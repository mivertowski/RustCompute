\usepackage{pgfplots}

\begin{document}

\begin{axis}[

    ybar,
    bar width=18pt,
    enlarge x limits=0.18,
    xlabel={Lifecycle rule},

    nodes near coords,
    nodes near coords style={font=\scriptsize, /pgf/number format/fixed,

    table[x=rule, y=mean_ns, y error=ci_halfwidth, col sep=comma]
